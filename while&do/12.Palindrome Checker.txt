import java.util.Scanner;

class Palindrome{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int org = sc.nextInt();
        int num = org;
        int rev=0;
        while(num!=0){
            int rem=num%10;
            rev=(rev*10)+rem;
            num=num/10;
        }
        if(org==rev){
            System.out.println(org+" is a Palindrome");
        }
        else{
            System.out.println(org+" is not a Palindrome.");
        }
    }
}
