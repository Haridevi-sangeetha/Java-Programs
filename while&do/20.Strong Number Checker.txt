import java.util.Scanner;

class StrongNumber{
    public static void main(String[]args){
        Scanner sc = new Scanner(System.in);
        int num=sc.nextInt();
        int temp=num;
        int fact;
        int sum=0;
        while(temp!=0){
            int rem=temp%10;
            fact=1;
            int i=1;
            while(i<=rem){
                fact=fact*i;
                i++;
            }
            sum=sum+fact;
            temp=temp/10;
        }
        if(sum==num){
            System.out.println(num+" is a strong number");
        }
        else{
            System.out.println(num+" is not a Strong number");
        }
    }
}
