import java.util.Scanner;

class PowerCalculator{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int base = sc.nextInt();
        int pow = sc.nextInt();
        int cal=1;
        int i=1;
        while(i<=pow){
            cal=cal*base;
            i++;
        }
        System.out.println(cal);
    }
}
