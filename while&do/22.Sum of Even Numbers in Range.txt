import java.util.Scanner;

class EvenSum{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int start=sc.nextInt();
        int end = sc.nextInt();
        int i=start;
        int sum=0;
        while(i<=end){
            if(i%2==0){
                sum=sum+i;
            }
            i++;
        }
        System.out.println("Sum of even numbers from "+start+" to "+end+" : "+sum);
    }
}
