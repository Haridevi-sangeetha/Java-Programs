import java.util.Scanner;

class PerfectNumber{
    public static void main(String [] args){
        Scanner sc=new Scanner(System.in);
        int num=sc.nextInt();
        int sum=0;
        int i=1;
        while(i<num){
            if(num%i==0){
                sum=sum+i;
            }
            i++;
        }
        if(sum==num){
            System.out.println(num+" is a Perfect number");
        }
        else{
            System.out.println(num+" is not a perfect number");
        }
    }
}
