import java.util.Scanner;

class FactorialCalculator{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int i=1;
        int fact = 1;
        while(i<=num){
            fact=fact*i;
            i++;
        }
        System.out.println("Factorial of "+num+" "+fact);
    }
}
