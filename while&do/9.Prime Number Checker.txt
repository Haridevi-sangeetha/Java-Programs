import java.util.Scanner;

class PrimeNumber{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int count=0;
        int i=1;
        while(i<=num){
            if(num%i==0){
                count++;
            }
            i++;
        }
        if(count==2){
            System.out.println(num+" Prime Number");
        }
        else{
            System.out.println(num+" Not Prime Number");
        }
    }
}
