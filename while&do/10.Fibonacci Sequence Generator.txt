import java.util.Scanner;

class Fibonacci{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int num=sc.nextInt();
        int first=0;
        int second=1;
        int i=0;
        while(i<num){
            System.out.print(first+" ");
            int temp = first;
            first=first+second;
            second=temp;
            i++;
        }

    }
}
