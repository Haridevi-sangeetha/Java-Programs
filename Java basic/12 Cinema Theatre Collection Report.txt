import java.util.Scanner;

class TheatreCollection{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int ticketRevenue = sc.nextInt();
        int snackRevenue  = sc .nextInt();
        int maintenanceCost = sc.nextInt();
        int electricityBil = sc.nextInt();

        int Profit  = ticketRevenue + snackRevenue - maintenanceCost - electricityBil;

        System.out.println("Net Profit : " + Profit);

    }
}
