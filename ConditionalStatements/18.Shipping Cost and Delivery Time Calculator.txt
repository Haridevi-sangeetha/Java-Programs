import java.util.Scanner;

class ShippingCost {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        double weight = sc.nextDouble();
        sc.nextLine(); 

        String zone = sc.nextLine();
        String speed = sc.nextLine();
        String size = sc.nextLine();

        double baseRate = 0;
        double weightSurcharge;
        double zoneMultiplier = 1.0;
        double sizeSurcharge = 0;
        int deliveryDays = 0;
        String serviceLevel = "";

        if (speed.equalsIgnoreCase("Economy")) {
            baseRate = 5;
            serviceLevel = "Budget";
        }
        else if (speed.equalsIgnoreCase("Standard")) {
            baseRate = 10;
            serviceLevel = "Standard";
        }
        else if (speed.equalsIgnoreCase("Express")) {
            baseRate = 25;
            serviceLevel = "Priority";
        }
        else if (speed.equalsIgnoreCase("Overnight")) {
            baseRate = 50;
            serviceLevel = "Premium";
        }

        
        weightSurcharge = weight;

        
        if (speed.equalsIgnoreCase("Overnight") &&
            zone.equalsIgnoreCase("International")) {
            weightSurcharge = weight * 2;
        }

        
        if (zone.equalsIgnoreCase("Local"))
            zoneMultiplier = 1.0;
        else if (zone.equalsIgnoreCase("Regional"))
            zoneMultiplier = 1.2;
        else if (zone.equalsIgnoreCase("National"))
            zoneMultiplier = 1.5;
        else if (zone.equalsIgnoreCase("International"))
            zoneMultiplier = 3.0;

        if (size.equalsIgnoreCase("Small"))
            sizeSurcharge = 0;
        else if (size.equalsIgnoreCase("Medium"))
            sizeSurcharge = 5;
        else if (size.equalsIgnoreCase("Large"))
            sizeSurcharge = 15;
        else if (size.equalsIgnoreCase("Oversized"))
            sizeSurcharge = 30;

        
        double baseCost = (baseRate + weightSurcharge) * zoneMultiplier;
        double totalCost = baseCost + sizeSurcharge;

        
        if (speed.equalsIgnoreCase("Economy")) {
            if (zone.equalsIgnoreCase("Local")) deliveryDays = 5;
            else if (zone.equalsIgnoreCase("Regional")) deliveryDays = 7;
            else if (zone.equalsIgnoreCase("National")) deliveryDays = 10;
            else deliveryDays = 21;
        }
        else if (speed.equalsIgnoreCase("Standard")) {
            if (zone.equalsIgnoreCase("Local")) deliveryDays = 3;
            else if (zone.equalsIgnoreCase("Regional")) deliveryDays = 5;
            else if (zone.equalsIgnoreCase("National")) deliveryDays = 7;
            else deliveryDays = 14;
        }
        else if (speed.equalsIgnoreCase("Express")) {
            if (zone.equalsIgnoreCase("Local")) deliveryDays = 1;
            else if (zone.equalsIgnoreCase("Regional")) deliveryDays = 2;
            else if (zone.equalsIgnoreCase("National")) deliveryDays = 2;
            else deliveryDays = 5;
        }
        else if (speed.equalsIgnoreCase("Overnight")) {
            deliveryDays = 1;
        }


        System.out.println("Package Weight: " + weight + " lbs");
        System.out.println("Destination Zone: " + zone);
        System.out.println("Shipping Speed: " + speed);
        System.out.println("Package Size: " + size);
        System.out.println("Base Shipping Rate: $" + baseRate);
        System.out.println("Weight Surcharge: $" + weightSurcharge);
        System.out.println("Zone Multiplier: " + zoneMultiplier + "x");
        System.out.println("Size Surcharge: $" + sizeSurcharge);
        System.out.println("Total Shipping Cost: $" + totalCost);
        System.out.println("Estimated Delivery: " + deliveryDays + " business days");
        System.out.println("Service Level: " + serviceLevel);
    }
}
