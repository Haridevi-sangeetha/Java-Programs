import java.util.Scanner;

class EmployeeSalary{
    public static void main(String [] args){
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        double sum = 0.0;
        System.out.println("");
        for(int i=1;i<=N;i++){
            String Name= sc.next();
            double Salary = sc.nextDouble();
            int Rating = sc.nextInt();
            int incremwnt = 0;
            if(Rating==5){
                incremwnt = 15;
            }
            else if(Rating == 4){
                incremwnt = 10;
            }
            else if(Rating == 3){
                incremwnt =5;
            }
            else if(Rating == 2){
                incremwnt = 2;
            }
            else if(Rating == 1){
                incremwnt = 0;
            }
            double BaseSalary = Salary+(Salary*incremwnt/100);
            System.out.println("Employee: "+Name);
            System.out.println("Base Salary: $"+Salary);
            System.out.println("Performance Rating: "+Rating);
            System.out.println("Increment: "+incremwnt+"%");
            System.out.println("Final Salary: $"+BaseSalary);

            System.out.println();
            sum = sum+BaseSalary;
        }
        System.out.println("Total Employees Processed: "+N);
        System.out.println("Total Payroll: $"+sum);
        double avg = sum/N;
        System.out.printf("Average Salary :$"+"%.2f",avg);
    }
}
